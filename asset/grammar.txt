<Program> = <StatementSequence>
<StatementSequence> = <Statement> | <StatementSequence> <Statement>  
<Statement> = $int <ID> <TypeStatement> | $void <ID> <FunctionStatement>
<TypeStatement> = <VariableStatement> | <FunctionStatement>
<VariableStatement> = $;
<FunctionStatement> = $( <Parameter> $) <SentenceBlock>
<Parameter> = <ParmList> | $void
<ParmList> = <Parm> | <Parm> $, <ParmList>
<Parm> = $int <ID> 
<SentenceBlock> = ${ <InternalStatement> <SentenceString> $}
<InternalStatement> = @empty | <InternalVarStatement> | <InternalVarStatement> $; <InternalStatement>
<InternalVarStatement> = $int <ID>
<SentenceString> = <Sentence> <SentenceString> | <Sentence>
<Sentence> = <if-Sentence> | <while-Sentence> | <return-Sentence> | <assign-Sentence>
<assign-Sentence> = <ID> $= <Expression> $;
<return-Sentence> = $return | $return <Expression>
<while-Sentence> = $while $( <Expression> $) <SentenceBlock>
<if-Sentence> = $if $( <Expression> $) <SentenceBlock> | $if $( <Expression> $) <SentenceBlock> $else <SentenceBlock>
<Expression> = <add-Expression> | <add-Expression> $relop <add-Expression> 
<add-Expression> = <Item> | <Item> $+ <add-Expression> | <Item> $- <add-Expression>
<Item> = <Factor> | <Factor> $* <Item> | <Factor> $/ <Item>
<Factor> = $num | $( <Expression> $) | <ID> <FTYPE>
<FTYPE> = <call> | @empty
<call> = $( <ArgumentList> $)
<Argument> = <ArgumentList> | @empty
<ArgumentList> = <Expression> | <Expression> $, <ArugmentList>
<ID> = $character <ID-temp>
<ID-temp> = $character <ID-temp> | $num <ID-temp> | @empty
